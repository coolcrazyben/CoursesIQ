Write the complete file as follows:

```typescript
'use client'

import { useState } from 'react'

type Alert = {
  id: string
  crn: string
  subject: string
  course_number: string
  course_name: string | null
  created_at: string
}

interface Props {
  alerts: Alert[]
}

export default function DashboardAlerts({ alerts: initial }: Props) {
  const [alerts, setAlerts] = useState<Alert[]>(initial)
  const [confirmingId, setConfirmingId] = useState<string | null>(null)
  const [cancelError, setCancelError] = useState<string | null>(null)

  async function handleCancel(id: string) {
    if (confirmingId !== id) {
      // First click: arm the confirmation
      setConfirmingId(id)
      setCancelError(null)
      return
    }

    // Second click: optimistic removal then fire PATCH
    setConfirmingId(null)
    setAlerts((prev) => prev.filter((a) => a.id !== id))

    try {
      const res = await fetch(`/api/alerts/${id}`, { method: 'PATCH' })
      if (!res.ok) {
        // Rollback: restore original alerts list
        setAlerts(initial)
        setCancelError('Failed to cancel alert. Please try again.')
      }
    } catch {
      // Network error: rollback
      setAlerts(initial)
      setCancelError('Network error. Please check your connection.')
    }
  }

  function handleCancelClick(id: string) {
    // If user clicks a different alert's cancel button, reset confirmation
    if (confirmingId !== null && confirmingId !== id) {
      setConfirmingId(null)
    }
    handleCancel(id)
  }

  if (alerts.length === 0) {
    return (
      <p className="text-gray-500 text-center py-8">
        No active alerts found for this number.
      </p>
    )
  }

  return (
    <div>
      {cancelError && (
        <p role="alert" className="text-red-600 text-sm mb-4">
          {cancelError}
        </p>
      )}
      <ul className="space-y-3">
        {alerts.map((alert) => {
          const courseName = alert.course_name ?? `${alert.subject} ${alert.course_number}`
          const isConfirming = confirmingId === alert.id
          return (
            <li
              key={alert.id}
              className="flex flex-col sm:flex-row sm:items-center sm:justify-between gap-3 bg-white border border-gray-200 rounded-lg p-4"
            >
              <div>
                <p className="font-semibold text-gray-900">{courseName}</p>
                <p className="text-sm text-gray-500">CRN: {alert.crn}</p>
              </div>
              <button
                onClick={() => handleCancelClick(alert.id)}
                className={`shrink-0 px-4 py-2 rounded-lg text-sm font-medium text-white transition-colors ${
                  isConfirming
                    ? 'bg-red-600 hover:bg-red-700'
                    : 'bg-maroon hover:opacity-90'
                }`}
              >
                {isConfirming ? 'Confirm cancel?' : 'Cancel alert'}
              </button>
            </li>
          )
        })}
      </ul>
    </div>
  )
}
```

- Empty state: renders "No active alerts found" paragraph (not an empty list)

```typescript
import { parsePhoneNumber } from 'libphonenumber-js'
import { adminClient } from '@/lib/supabase/admin'
import DashboardAlerts from '@/components/DashboardAlerts'

interface PageProps {
  searchParams: Promise<{ phone?: string }>
}

type Alert = {
  id: string
  crn: string
  subject: string
  course_number: string
  course_name: string | null
  created_at: string
}

async function fetchAlertsForPhone(rawPhone: string): Promise<Alert[]> {
  let e164: string
  try {
    const parsed = parsePhoneNumber(rawPhone, 'US')
    if (!parsed || !parsed.isValid()) return []
    e164 = parsed.number
  } catch {
    return []
  }

  const { data, error } = await adminClient
    .from('alerts')
    .select('id, crn, subject, course_number, course_name, created_at')
    .eq('phone_number', e164)
    .eq('is_active', true)
    .order('created_at', { ascending: false })

  if (error) {
    console.error('[dashboard] Supabase fetch error:', error.message)
    return []
  }

  return data ?? []
}

export default async function DashboardPage({ searchParams }: PageProps) {
  // Next.js 15: searchParams is a Promise — MUST await
  const { phone } = await searchParams

  const alerts: Alert[] = phone ? await fetchAlertsForPhone(phone) : []
  const hasSearched = Boolean(phone)

  return (
    <main className="max-w-2xl mx-auto px-4 py-10">
      <h1 className="text-2xl font-bold text-maroon mb-2">My Alerts</h1>
      <p className="text-gray-600 mb-8">
        Enter your phone number to view and manage your active seat alerts.
      </p>

      {/* Phone lookup form — plain GET form, no JavaScript required */}
      <form method="GET" action="/dashboard" className="flex gap-2 mb-8">
        <input
          type="tel"
          name="phone"
          defaultValue={phone ?? ''}
          placeholder="(601) 555-1234"
          required
          className="flex-1 px-3 py-3 border border-gray-300 rounded-lg text-base focus:outline-none focus:ring-2 focus:ring-maroon focus:border-maroon"
        />
        <button
          type="submit"
          className="bg-maroon text-white px-5 py-3 rounded-lg text-sm font-semibold hover:opacity-90 transition-opacity whitespace-nowrap"
        >
          Look up
        </button>
      </form>

      {/* Results section */}
      {hasSearched && (
        <section>
          <h2 className="text-lg font-semibold text-gray-800 mb-4">
            Active alerts for{' '}
            <span className="text-maroon">{phone}</span>
          </h2>
          <DashboardAlerts alerts={alerts} />
        </section>
      )}
    </main>
  )
}
```

Key implementation notes:
- `const { phone } = await searchParams` — the `await` is MANDATORY (Next.js 15)
- `fetchAlertsForPhone` normalizes the raw URL phone to E.164 before querying — prevents zero-result bug when user types "601-555-1234" but stored value is "+16015551234"
- `parsePhoneNumber` returns `[]` on invalid input — dashboard never throws from a bad phone
- The `<form method="GET" action="/dashboard">` causes a full-page navigation that re-runs the server component with fresh searchParams — no useState or router.push needed
- `defaultValue={phone ?? ''}` pre-fills the input with the current lookup phone so users can see/edit it
- `hasSearched` guards the results section — no results shown until the form is submitted
- `DashboardAlerts alerts={alerts}` passes the server-fetched array as a prop to the client component
- The `Alert` type is declared in both files — they are not imported from a shared types file to keep each file self-contained (no shared types module exists in this project yet)

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → /dashboard?phone=... | The `phone` query parameter is user-controlled input arriving via URL |
| DashboardAlerts → PATCH /api/alerts/[id] | The `id` path segment in the cancel fetch is taken from the server-rendered alert data |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-04-01 | Spoofing | Dashboard phone lookup | accept | Anyone who knows a phone number can look up that number's alerts. This is an accepted v1 tradeoff — zero-friction is a locked decision (CONTEXT.md). The UI shows this clearly (no false sense of security). Auth deferred to v2 (AUTH-01). |
| T-04-04-02 | Tampering | searchParams phone injection | mitigate | `parsePhoneNumber('US')` rejects non-phone strings (e.g., SQL injection attempts). Invalid input returns `[]` from `fetchAlertsForPhone` — no query is run. |
| T-04-04-03 | Information Disclosure | Dashboard alert list | accept | Alert data (course name, CRN) is non-sensitive course registration information. Exposing it to anyone with the phone number is the accepted v1 tradeoff per CONTEXT.md. |
| T-04-04-04 | Tampering | DashboardAlerts cancel — alert ID from props | accept | Alert IDs come from server-rendered data (not user input). While PATCH /api/alerts/[id] accepts any UUID, the IDs displayed are the user's own alerts from the Supabase query. Unauthorized cancellation risk is mitigated by ID non-enumerability (UUIDs). |
| T-04-04-05 | XSS | DashboardAlerts — course_name display | mitigate | `course_name` and `crn` values are rendered as React text nodes — not `dangerouslySetInnerHTML`. React escapes all dynamic string values automatically. The `??` fallback to `subject + course_number` also only uses React text rendering. |
| T-04-04-06 | Denial of Service | Dashboard query — bulk phone scan | accept | No rate limiting in v1. An attacker could query many phone numbers. Queries return only non-PII course data. Acceptable risk for v1 student project; rate limiting deferred to v2. |
</threat_model>
