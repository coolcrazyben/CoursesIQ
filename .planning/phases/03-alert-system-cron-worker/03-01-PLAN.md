```typescript
import 'server-only'
// IMPORTANT: Any API route importing this file must declare:
//   export const runtime = 'nodejs'
// Twilio requires Node.js native modules; Edge Runtime is not supported.
import twilio from 'twilio'

// Module-level singleton — one Twilio client instance per cold start.
// Mirrors the lib/banner.ts module-level client pattern.
const twilioClient = twilio(
  process.env.TWILIO_ACCOUNT_SID!,
  process.env.TWILIO_AUTH_TOKEN!
)

/**
 * Send a seat-open SMS alert to a subscriber.
 *
 * @param phone - E.164 phone number (e.g., '+16015551234')
 * @param courseName - Human-readable course name (e.g., 'Intro to Computer Science')
 * @param crn - Course Reference Number (e.g., '31352')
 * @returns The Twilio message SID (e.g., 'SMxxxxxxxxxxxxxxxxxxxxxxxxxxxx')
 * @throws RestException on Twilio API errors — callers are responsible for catching
 *   error code 21610 (opted out) and setting sms_opted_out = true
 */
export async function sendSeatAlert(
  phone: string,
  courseName: string,
  crn: string
): Promise<string> {
  const message = await twilioClient.messages.create({
    body: `🎉 A seat just opened in ${courseName} (${crn})! Register now before it fills up: mybanner.msstate.edu — CoursesIQ`,
    from: process.env.TWILIO_PHONE_NUMBER!,
    to: phone,
  })
  return message.sid
}
```

Implementation notes:
- `import 'server-only'` MUST be line 1 — prevents client-side import at build time
- `import twilio from 'twilio'` — default import, called as a function (not `new twilio(...)`)
- The SMS body must use the exact emoji (U+1F389) and match ALRT-08 format exactly:
  `🎉 A seat just opened in ${courseName} (${crn})! Register now before it fills up: mybanner.msstate.edu — CoursesIQ`
- `from` uses `TWILIO_PHONE_NUMBER` env var — not hardcoded
- The function throws on all errors including error 21610; callers handle error dispatch
- No try/catch inside sendSeatAlert — error propagates to the cron route caller

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-04 | Disclosure | TWILIO_ACCOUNT_SID / TWILIO_AUTH_TOKEN in process.env | mitigate | `import 'server-only'` in lib/twilio.ts causes build-time error if imported in Client Component; env vars never prefixed NEXT_PUBLIC_; .env.local is gitignored |
| T-03-05 | Tampering | SMS body injection via courseName or crn params | accept | Both values originate from Supabase alerts table (developer-controlled insert) or Banner API (trusted source); no user-controlled input reaches the SMS body at dispatch time |
| T-03-06 | Denial of Service | Twilio rate limits on message creation | accept | One SMS per seat-opening event per alert (sms_sent_at gate prevents repeat sends); cron runs every 5 minutes — not a burst pattern |
| T-03-07 | Elevation | Migration adds UNIQUE constraint — could fail if duplicate rows already exist | mitigate | If migration fails, executor must run `DELETE FROM public.alerts WHERE id NOT IN (SELECT MIN(id) FROM public.alerts GROUP BY crn, phone_number)` before retrying the ALTER TABLE |
</threat_model>
